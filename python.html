🔹 Example: Simple Class & Inheritance

class Student:
    def __init__(self, name, roll):
        self.name = name
        self.roll = roll
    
    def display(self):
        print(f"Student: {self.name} ({self.roll})")

class BSCPEStudent(Student):
    def __init__(self, name, roll, cgpa):
        super().__init__(name, roll)
        self.cgpa = cgpa
